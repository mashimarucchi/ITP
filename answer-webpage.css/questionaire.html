        import java.util.Scanner;

public class RangeDetector {
   public static void main(String[] args) {
      Scanner scnr = new Scanner(System.in);
      int x;

      x = scnr.nextInt();

      if (x < 15) {
         System.out.println("n");
      }
      else if (x < 30) {
         System.out.println("k");
      }
      else if (x < 35) {
         System.out.println("s");
      }
      else {
         System.out.println("z");
      }
   }
}
